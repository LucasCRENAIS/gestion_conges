// Fetch school vacations for zone C (Toulouse) from the French Education Ministry API function getSchoolVacations() { const today = new Date(); const currentYear = today.getFullYear(); const lastUpdateDate = new Date(localStorage.getItem('vacationsLastUpdate')); // Check if it's July 1st of the current year or if it's the first call if ((today.getMonth() === 6 && today.getDate() === 1) || lastUpdateDate.getFullYear() !== currentYear) { fetch(`https://cache.education.gouv.fr/education/gouv/jes/sv/${currentYear}/vacancesZoneC.json`) .then(response => response.json()) .then(data => { // Store the fetched vacations in localStorage localStorage.setItem('schoolVacationsZoneC', JSON.stringify(data)); localStorage.setItem('vacationsLastUpdate', today.toISOString()); }) .catch(error => console.error('Error fetching school vacations:', error)); } else { console.log('Using cached school vacations data.'); } }
